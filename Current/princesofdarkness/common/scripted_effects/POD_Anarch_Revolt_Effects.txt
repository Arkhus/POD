pod_break_bloodbond = {
	if = {
		limit = {
			trait = bloodbound
		}
		save_event_target_as = target_bound_person
		any_opinion_modifier_target = {
			limit = {
				has_opinion_modifier = {
					who = event_target:target_bound_person
					modifier = opinion_blood_bond_to_me
				}
			}
			log = "[This.GetTitledName] is domitor to [target_bound_person.GetTitledName], now broken"
			save_event_target_as = domitor_target
			event_target:target_bound_person = {
				remove_opinion = {
					who = event_target:domitor_target
					modifier = opinion_blood_bond_to_domitor
				}
			}
			remove_opinion = {
				who = event_target:target_bound_person
				modifier = opinion_blood_bond_to_me
			}				
		}
		remove_trait = bloodbound
	}
}

anarch_problem_province_effect = {
	set_variable = { which = anarch_count value = 5 }
	trigger_switch = {
		on_trigger = has_global_flag
		anarch_revolt_stage_1 = {
			set_variable = { which = anarch_count value = 5 }
		}
		anarch_revolt_stage_2 = {
			set_variable = { which = anarch_count value = 12 }
		}
		anarch_revolt_stage_3 = {
			set_variable = { which = anarch_count value = 24 }
		}
	}
	while = {
		limit = {
			check_variable = {
				which = anarch_count
				value = 1
			}
		}    
		subtract_variable = {
			which = anarch_count
			value = 1
		}
		random_province = {
			limit = {
				NOT = { has_province_modifier = anarch_problem_1_modifier }
				NOT = { has_province_modifier = anarch_problem_2_modifier }
				NOT = { has_province_modifier = anarch_problem_3_modifier }
				owner = { 
					trait = vampire
					NOT = { trait = storyteller }
					is_vampire_anarch = no
				}
			}
			
			random_list = {
				5 = {
					modifier = {
						factor = 5
						has_global_flag = anarch_revolt_stage_1
					}
					add_province_modifier = {
						name = anarch_problem_1_modifier
						years = 10
					}
				}
				5 = {
					modifier = {
						factor = 4
						has_global_flag = anarch_revolt_stage_2
					}
					add_province_modifier = {
						name = anarch_problem_2_modifier
						years = 10
					}
				}
				5 = {
					modifier = {
						factor = 3
						has_global_flag = anarch_revolt_stage_3
					}
					add_province_modifier = {
						name = anarch_problem_3_modifier
						years = 10
					}
				}
			}
		}
		change_variable = { which = global_anarch_revolt_counter value = 1 }
	}
}

check_anarch_revolt_stage_effect = {
	log = "checking anarch revolt stage"
	log = "Anarch Revolt Counter: [global_anarch_revolt_counter.GetValue]"
	log = "Minor Anarch Uprisings: [global_anarch_minor_revolt_count.GetValue]"
	log = "Major Anarch Uprisings: [global_anarch_major_revolt_count.GetValue]"
	if = {
		limit = {
			check_variable = { which = "global_anarch_revolt_counter" value = 20 }
			NOT = { has_global_flag = first_brujah_adventurer }
		}
		log = "firing brujah anarch adventurer"
		set_global_flag = first_brujah_adventurer
		pod_fire_brujah_anarch_1 = yes
		any_player = {
			narrative_event = { id = POD_Anarch_Revolt_Events.204 days = 3 } 
		}
	}else_if = {
		limit = {
			check_variable = { which = "global_anarch_revolt_counter" value = 200 }
			NOR = { 
				has_global_flag = anarch_revolt_stage_2
				has_global_flag = anarch_revolt_stage_3
				has_global_flag = anarch_revolt_stage_4
			}
		}
		set_global_flag = anarch_revolt_stage_2
		clr_global_flag = anarch_revolt_stage_1
		log = "setting anarch revolt stage 2"
		set_variable = { which = global_anarch_revolt_level value = 2 }
		any_player = {
			narrative_event = { id = POD_Anarch_Revolt_Events.101 days = 10 }
			narrative_event = { id = POD_Anarch_Revolt_Events.205 days = 30 random = 30 }
		}
	}else_if = {
		limit = {
			check_variable = { which = "global_anarch_revolt_counter" value = 680 }
			NOR = { 
				has_global_flag = anarch_revolt_stage_3
				has_global_flag = anarch_revolt_stage_4
			}
		}
		set_global_flag = anarch_revolt_stage_3
		clr_global_flag = anarch_revolt_stage_2
		log = "setting anarch revolt stage 3"
		set_variable = { which = global_anarch_revolt_level value = 3 }
		any_player = {
			narrative_event = { id = POD_Anarch_Revolt_Events.102 }
		}
		log = "firing wide anarch conversion"
		province_event = { id = POD_Anarch_Revolt_Events.212 days = 30 random = 60 }
	}else_if = {
		limit = {
			check_variable = { which = "global_anarch_revolt_counter" value = 920 }
			NOT = { has_global_flag = sabbat_usurper_1_fired }
		}
		log = "firing sabbat usurper 1"
		province_event = { id = POD_Anarch_Revolt_Events.210 days = 30 random = 60 }
	}else_if = {
		limit = {
			check_variable = { which = "global_anarch_revolt_counter" value = 1160 }
			NOT = { has_global_flag = sabbat_usurper_2_fired }
		}
		log = "firing sabbat usurper 2"
		province_event = { id = POD_Anarch_Revolt_Events.211 days = 30 random = 60 }
	}else_if = {
		limit = {
			check_variable = { which = "global_anarch_revolt_counter" value = 1400 }
			NOR = { 
				has_global_flag = anarch_revolt_stage_4
			}
		}
		set_variable = { which = global_anarch_revolt_level value = 4 }
		set_global_flag = anarch_revolt_stage_4
		set_global_flag = anarch_revolt_end
		clr_global_flag = anarch_revolt_stage_3
		log = "setting anarch revolt stage 4"
		province_event = { id = POD_Camarilla_Sabbat_Formation_Events.1 days = 30 random = 260 }
		any_player = {
			narrative_event = { id = POD_Anarch_Revolt_Events.103 days = 3 }
		}
	}

		log = "CAMARILLA FOUNDER IS PLAYER"
	}else = {
		log = "CAMARILLA FOUNDER IS NOT PLAYER"
	}
	set_character_flag = CAMARILLA_EMPEROR_PERSON
}

pod_check_target_size_effect = {
	export_to_variable = { which = local_new_size value = realm_size }
			if = {
				limit = {
					NOT = { check_variable = { which = global_sabbat_size which = local_new_size } }
				}
				log = "changed largest ruler: [local_new_size.GetValue] [This.GetTitledName]"
				set_variable = { which = global_sabbat_size which = local_new_size }
				save_event_target_as = biggest_ruler
			}
}

		random_character = {
			limit = {
				is_ruler = no
				trait = vampire
				trait = assamite
			}
			save_event_target_as = anarch_adventurer
			log = "[anarch_adventurer.GetTitledName] is 1 choice"
		}
		random_character = {
			limit = {
				is_ruler = no
				trait = vampire
				trait = assamite
				martial = 20
			}
			save_event_target_as = anarch_adventurer
			log = "[anarch_adventurer.GetTitledName] is 2 choice"
		}
		random_character = {
			limit = {
				is_ruler = no
				trait = vampire
				trait = assamite
				trait = canon_childe
				age < 401
				not_likely_anarch = no
			}
			save_event_target_as = anarch_adventurer
			log = "[anarch_adventurer.GetTitledName] is 3 choice"
		}
		if = {
			limit = {
				is_late_anarch_revolt = yes
			}
		random_character = {
			limit = {
				is_ruler = no
				trait = vampire
				trait = assamite
				trait = canon
				age < 401
				not_likely_anarch = no
			}
			save_event_target_as = anarch_adventurer
			log = "[anarch_adventurer.GetTitledName] is 4 choice"
		}
		random_character = {
			limit = {
				is_ruler = no
				trait = vampire
				trait = assamite
				trait = great_person
				has_character_flag = great_rebel
				age < 401
				not_likely_anarch = no
			}
			save_event_target_as = anarch_adventurer
			log = "[anarch_adventurer.GetTitledName] is 5 choice"
		}
		}
		if = {
			limit = { NOT = { event_target:anarch_adventurer = { always = yes } } }
			break = yes
		}
		event_target:anarch_adventurer = {
			set_character_flag = anarch_revolter
			set_character_flag = anarch_adventurer
			set_character_flag = made_into_anarch
			create_anarch_adventurer_titles = yes
			#add_trait = anarch_revolt_figure_trait
		}
}
