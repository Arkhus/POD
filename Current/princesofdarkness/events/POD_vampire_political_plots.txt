	id = POD_vampire_political_plots.1
	hide_window = yes
	
	only_playable = yes
	ai = no
	capable_only = yes
	prisoner = no
	is_part_of_plot = yes
	
	trigger = {
		ai = no
		OR = {
			has_plot = plot_seed_of_rebellion
			has_plot = plot_frame_crime
			has_plot = plot_awaken_target
			has_plot = plot_bloodbind_target
			has_plot = plot_kidnap
			has_plot = plot_scheme_friendship
			has_plot = plot_scheme_loyalty
		}
		plot_target_char = {
			ai = yes
			NOT = { is_murder_quest_target_of_root_trigger = yes }
			NOT = { trait = incapable }
			prisoner = no
			AND = {
				NOT = { is_inaccessible_trigger = yes }
				NOT = { ROOT = { has_plot = plot_awaken_target } }
			}
			AND = {
				NOT = { trait = torpor }
				NOT = { ROOT = { has_plot = plot_awaken_target } }
			}
			
		}
		plot_power_contribution = 
		{
			plotter = ROOT
			power = 0.025
		}
		NOT = { has_character_flag = political_plot_ongoing }
	}
	
	mean_time_to_happen = {
		months = 800
		modifier = {
			factor = 0.5
			plot_power = 0.5
		}
		modifier = {
			factor = 0.5
			plot_power = 1.0
		}
		modifier = {
			factor = 0.4
			plot_power = 1.5
		}
		modifier = {
			factor = 0.4
			plot_power = 2.0
		}
		modifier = {
			factor = 0.6
			intrigue = 10
		}
		modifier = {
			factor = 0.6
			intrigue = 20
		}
		modifier = {
			factor = 0.5
			intrigue = 30
		}
		modifier = {
			factor = 0.4
			intrigue = 40
		}
		modifier = {
			factor = 0.1
			has_plot = plot_awaken_target
			plot_target_char = {
				is_friend = ROOT
			}
		}
	}
	
	immediate = {
		plot_target_char = { save_event_target_as = pol_target }
		save_event_target_as = pol_plot_creator
		trigger_switch = {
			on_trigger = has_plot
			plot_scheme_loyalty = {
				character_event = { id = POD_vampire_political_plots.2 }
			}
			plot_scheme_friendship = {
				character_event = { id = POD_vampire_political_plots.4 }
			}
			plot_seed_of_rebellion = {
				character_event = { id = POD_vampire_political_plots.6 }

				character_event = { id = POD_vampire_political_plots.8 }
