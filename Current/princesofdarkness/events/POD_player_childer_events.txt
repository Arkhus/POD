#Player Ping for Childe Events
character_event = {
    id = POD_player_childe_events.1
	hide_window = yes
	only_playable = yes
	ai = no
    is_triggered_only = yes

	
	trigger = {
		is_alive = yes
		trait = vampire
	}
	
	#Search player for childe related events and flags
    immediate = {
		save_event_target_as = player_parent
		any_child = {
			limit = {
				NOT = { has_character_flag = CHILDE_SPREAD_WINGS }
			}
			character_event = { id = POD_player_childe_events.2 days = 1 random = 80 }
		}
		random = {
			chance = 6
			character_event = { id = POD_player_childe_events.34 days = 60 random = 90 }
		}
	}
}

#Childe Event Controller
character_event = {
    id = POD_player_childe_events.2
	
    is_triggered_only = yes
	hide_window = yes

    immediate = {
		save_event_target_as = player_childe
		if = {
			limit = { 
				NOT = { has_character_flag = MENTOR_DECISION_TAKEN }
				age < 120
			}
			set_character_flag = MENTOR_DECISION_TAKEN
			event_target:player_parent = {
				character_event = { id = POD_player_childe_events.6 }
			}
		}else_if = {
			limit = { has_character_flag = IS_PLAYER_MENTEE }
			if = {
				limit = {
					OR = {
						has_character_flag = MENTORSHIP_DONE
						trait = ancilla
						trait = elder
						trait = ancient
						trait = methuselah
						NOT = { any_liege = { character = event_target:player_parent } }
						trait = torpor
					}
				}
				if = {
					limit = { has_character_flag = MENTORSHIP_DONE }
					clr_character_flag = MENTORSHIP_DONE
				}
				clr_character_flag = IS_PLAYER_MENTEE
				#notify player childe is no longer mentored
				event_target:player_parent = {
					character_event = { id = POD_player_childe_events.7 }
				}
			}else = {
				#mentee events
				character_event = { id = POD_player_childe_events.8 }
			}
		}
		if = {
			limit = {
				OR = {
					age = 120
					NOT = { any_liege = { character = event_target:player_parent } }
				}
			}
			hidden_tooltip = {
				set_character_flag = CHILDE_SPREAD_WINGS		
			}
		}else = {
			if = {
				limit = {
					NOT = { has_character_flag = REACTION_TO_UNLIFE }
					NOT = { trait = torpor }
					had_character_flag = { flag = MENTOR_DECISION_TAKEN days = 1200 }
				}
				set_character_flag = REACTION_TO_UNLIFE
				event_target:player_parent = { character_event = { id = POD_player_childe_events.18 days = 1 random = 20 } }
			}
			if = {
				limit = {
					NOT = { has_character_flag = REACTION_TO_SIRE }
					NOT = { trait = torpor }
					had_character_flag = { flag = MENTOR_DECISION_TAKEN days = 4500 }
				}
				set_character_flag = REACTION_TO_SIRE
				event_target:player_parent = { character_event = { id = POD_player_childe_events.19 days = 1 random = 20 } }
			}
			if = {
				limit = { NOT = { trait = torpor } }
			random_list = {
				3 = {
					character_event = { id = POD_player_childe_events.3 }
				}
				5 = {
					character_event = { id = POD_player_childe_events.4 }
				}
				13 = {
					character_event = { id = POD_player_childe_events.5 }
				}
				180 = {
					#nothing
				}
			}
			}
		}
	}
}

#Mentee Events Controller
character_event = {
    id = POD_player_childe_events.8
	
    is_triggered_only = yes
	hide_window = yes

    immediate = {
		if = {
			limit = { NOT = { has_character_flag = mentor_basics_taught } }
			set_character_flag = mentor_basics_taught
			
			#teach basics event
			event_target:player_parent = {
				character_event = { id = POD_player_childe_events.9 days = 1 random = 4 }
			}
		}else_if = {
			limit = { 
				had_character_flag = { flag = mentor_basics_taught days = 730 }
				NOT = { has_character_flag = mentor_vampire_taught }
			}
			set_character_flag = mentor_vampire_taught
			event_target:player_parent = {
				character_event = { id = POD_player_childe_events.10 days = 1 random = 4 }
			}
		}else_if = {
			limit = { 
				had_character_flag = { flag = mentor_vampire_taught days = 730 }
				NOT = { has_character_flag = mentor_court_things }
			}
			set_character_flag = mentor_court_things
			event_target:player_parent = {
				character_event = { id = POD_player_childe_events.11 days = 1 random = 4 }
			}
		}else_if = {
			limit = { 
				had_character_flag = { flag = mentor_court_things days = 730 }
				NOT = { has_character_flag = mentor_path_things }
			}
			set_character_flag = mentor_path_things
			event_target:player_parent = {
				character_event = { id = POD_player_childe_events.12 days = 1 random = 4 }
			}
		}else_if = {
			limit = { 
				had_character_flag = { flag = mentor_path_things days = 730 }
				NOT = { has_character_flag = MENTORSHIP_DONE }
			}
			set_character_flag = MENTORSHIP_DONE
			clr_character_flag = IS_PLAYER_MENTEE
			#notify player childe is no longer mentored
			event_target:player_parent = {
				character_event = { id = POD_player_childe_events.7 }
			}
		}
		random_list = {
			3 = {
				#mentee crisis
				modifier = {
					factor = 0
					has_character_flag = had_mentee_crisis
				}
				set_character_flag = had_mentee_crisis
				event_target:player_parent = {
					character_event = { id = POD_player_childe_events.13 days = 1 random = 25 }
				}
			}
			9 = {
				#mentee gains a trait of yours
				event_target:player_parent = {
					character_event = { id = POD_player_childe_events.14 days = 1 random = 25 }
				}
			}
			3 = {
				modifier = {
					factor = 0
					is_friend = event_target:player_parent
				}
				modifier = {
					factor = 2
					opinion = { who = event_target:player_parent value = 90 }
					event_target:player_parent = {
						opinion = { who = ROOT value = 70 }
					}
				}
				event_target:player_parent = {
					character_event = { id = POD_player_childe_events.15 days = 1 random = 25 }
				}
				#mentee becomes your friend
			}
			5 = {
				modifier = {
					factor = 0
					has_character_flag = developed_feelings
				}
				set_character_flag = developed_feelings
				event_target:player_parent = {
					character_event = { id = POD_player_childe_events.16 days = 1 random = 25 }
				}
				#mentee develops strong feelings towards you
			}
			2 = {
				modifier = {
					factor = 4
					NOT = { opinion = { who = event_target:player_parent value = 0 } } 
				}
				modifier = {
					factor = 0.1
					opinion = { who = event_target:player_parent value = 50 }
				}
				event_target:player_parent = {
					character_event = { id = POD_player_childe_events.17 days = 1 random = 25 }
				}
				#mentee becomes rival with you and leaves court
			}
			80 = {
				#no event
			}
		}
	}
}
